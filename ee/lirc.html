# 配置设备树

修改 /boot/config.txt 文件，开启 gpio-ir 内核模块

dtoverlay=gpio-ir,gpio_pin=22
dtoverlay=gpio-ir-tx,gpio_pin=23
1
2

保存，重启，检查系统启动日志

pi@idf-morris:~ $ dmesg | grep lirc
[    4.233055] rc rc1: lirc_dev: driver gpio-ir-tx registered at minor = 0, no receiver, raw IR transmitter
[    4.292858] rc rc0: lirc_dev: driver gpio_ir_recv registered at minor = 1, raw IR receiver, no transmitter
1
2
3

# 安装 lirc

sudo apt-get install lirc
1

# 修改 lirc_options.conf

sudo mv lirc_options.conf.dist lirc_options.conf
1

修改 lirc_options.conf 其中的 driver 和 device 字段为如下：

[lircd]
driver = default
device = /dev/lirc
1
2
3

# 验证是否安装成功

pi@idf-morris:/etc/lirc $ lircd -v
lircd 0.10.1
1
