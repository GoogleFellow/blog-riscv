# 流程控制中的 break 和 continue

scala 语言中没有 break 和 continue 关键字，但是可以通过异常捕获或者循环守卫的方式来达到类似效果。

# breakable

import scala.util.control.Breaks.{break, breakable}

// break
breakable {
    for (i <- 1 to 10) {
        if (i == 8) {
            break()
        }
        println("i = " + i)
    }
}

// continue
for (i <- 1 to 10) {
    breakable {
        if (i == 3 || i == 8) {
            break()
        }
        println("i = " + i)
    }
}

# 循环守卫

// break
var loop = true
for (i <- 1 to 10 if loop == true) {
	if (i == 8) {
		loop = false
	}
  	println("i = " + i)
}

// continue
for (i <- 1 to 10 if i != 8) {
	println("i = " + i)
}
